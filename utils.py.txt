def validate_date(date)
    Validate date format YYYY-MM-DD.
    import datetime
    try
        datetime.datetime.strptime(date, %Y-%m-%d)
        return True
    except ValueError
        return False

def validate_time(time)
    Validate time format HHMM (24-hour).
    import datetime
    try
        datetime.datetime.strptime(time, %H%M)
        return True
    except ValueError
        return False
