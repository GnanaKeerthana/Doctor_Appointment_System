class Appointment
    Represents an appointment between a patient and a doctor.

    def __init__(self, patient_id, doctor_id, date, time)
        self.patient_id = patient_id
        self.doctor_id = doctor_id
        self.date = date
        self.time = time

    def __str__(self)
        return fAppointment - Patient ID {self.patient_id}, Doctor ID {self.doctor_id}, Date {self.date}, Time {self.time}
